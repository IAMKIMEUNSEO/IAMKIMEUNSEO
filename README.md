```python
  # 각도 계산 함수
def calculate_angle(a, b, c):
    a = np.array(a)  # 좌표를 배열로 변환
    b = np.array(b)
    c = np.array(c)
    
    radians = np.arctan2(c[1] - b[1], c[0] - b[0]) - np.arctan2(a[1] - b[1], a[0] - b[0])  # 각도 계산
    angle = np.abs(radians * 180.0 / np.pi)  # 라디안을 각도로 변환
 ```
